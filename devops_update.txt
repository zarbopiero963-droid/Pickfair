        "pytest",
        "-v",
        "-p",
        "no:pytestqt",
    ]

    if targets:
        cmd.extend(targets)

    env = os.environ.copy()

    env["PYTEST_DISABLE_PLUGIN_AUTOLOAD"] = "1"

    result = subprocess.run(
        cmd,
        capture_output=True,
        text=True,
        env=env,
    )

    print(result.stdout, flush=True)
    print(result.stderr, flush=True)

    if result.returncode != 0:
        log("Tests FAILED")
        return False

    log("Tests PASSED")

    return True
