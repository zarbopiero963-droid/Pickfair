CREATE_FILE .github/scripts/generate_repo_autopsy.py import ast import csv import json import os import traceback from collections import Counter, defaultdict from pathlib import Path

IGNORE_DIRS = { ".git", ".github", ".venv", "venv", "pycache", ".pytest_cache", "build", "dist", }
def is_ignored(path: Path) -> bool: return any(part in IGNORE_DIRS for part in path.parts)
def discover_python_files(): return sorted( p for p in ROOT.rglob("*.py") if not is_ignored(p.relative_to(ROOT)) )
def safe_unparse(node): try: return ast.unparse(node) except Exception: return "<unparse_error>"
class RepoVisitor(ast.NodeVisitor): def init(self, relpath: str): self.relpath = relpath self.imports = [] self.from_imports = [] self.classes = [] self.functions = [] self.calls = [] self.class_edges = [] self.current_class = None self.current_function = None
def scope(self):
    if self.current_class and self.current_function:
        return f"{self.current_class}.{self.current_function}"
    if self.current_class:
        return self.current_class
    if self.current_function:
        return self.current_function
    return "<module>"

def visit_Import(self, node):
    for alias in node.names:
        self.imports.append({"module": alias.name, "lineno": node.lineno})
    self.generic_visit(node)

def visit_ImportFrom(self, node):
    mod = node.module or ""
    for alias in node.names:
        self.from_imports.append(
            {
                "module": mod,
                "name": alias.name,
                "level": node.level,
                "lineno": node.lineno,
            }
        )
    self.generic_visit(node)

def visit_ClassDef(self, node):
    prev = self.current_class
    self.current_class = node.name
    self.classes.append(
        {
            "name": node.name,
            "lineno": node.lineno,
            "bases": [safe_unparse(b) for b in node.bases],
        }
    )
    for base in node.bases:
        self.class_edges.append(
            {
                "src_file": self.relpath,
                "src_class": node.name,
                "dst_class": safe_unparse(base),
                "lineno": node.lineno,
            }
        )
    self.generic_visit(node)
    self.current_class = prev

def visit_FunctionDef(self, node):
    prev = self.current_function
    self.current_function = node.name
    self.functions.append(
        {
            "name": node.name,
            "scope": self.scope(),
            "lineno": node.lineno,
            "args": [a.arg for a in node.args.args],
            "kwonlyargs": [a.arg for a in node.args.kwonlyargs],
            "vararg": node.args.vararg.arg if node.args.vararg else None,
            "kwarg": node.args.kwarg.arg if node.args.kwarg else None,
        }
    )
    self.generic_visit(node)
    self.current_function = prev

def visit_AsyncFunctionDef(self, node):
    prev = self.current_function
    self.current_function = node.name
    self.functions.append(
        {
            "name": node.name,
            "scope": self.scope(),
            "lineno": node.lineno,
            "args": [a.arg for a in node.args.args],
            "kwonlyargs": [a.arg for a in node.args.kwonlyargs],
            "vararg": node.args.vararg.arg if node.args.vararg else None,
            "kwarg": node.args.kwarg.arg if node.args.kwarg else None,
            "async": True,
        }
    )
    self.generic_visit(node)
    self.current_function = prev

def visit_Call(self, node):
    self.calls.append(
        {
            "src_file": self.relpath,
            "scope": self.scope(),
            "call": safe_unparse(node.func),
            "lineno": node.lineno,
            "args_count": len(node.args),
            "kwargs_count": len(node.keywords),
        }
    )
    self.generic_visit(node)
def parse_file(path: Path): rel = path.relative_to(ROOT).as_posix() result = { "file": rel, "ok": True, "syntax_error": None, "traceback": None, "imports": [], "from_imports": [], "classes": [], "functions": [], "calls": [], "class_edges": [], "lines": 0, } try: text = path.read_text(encoding="utf-8") result["lines"] = len(text.splitlines()) tree = ast.parse(text, filename=rel) v = RepoVisitor(rel) v.visit(tree) result["imports"] = v.imports result["from_imports"] = v.from_imports result["classes"] = v.classes result["functions"] = v.functions result["calls"] = v.calls result["class_edges"] = v.class_edges except SyntaxError as e: result["ok"] = False result["syntax_error"] = { "msg": e.msg, "lineno": e.lineno, "offset": e.offset, "text": e.text, } except Exception: result["ok"] = False result["traceback"] = traceback.format_exc() return result
def write_json(path: Path, data): path.write_text(json.dumps(data, indent=2, ensure_ascii=False), encoding="utf-8")
def write_csv(path: Path, rows): if not rows: path.write_text("", encoding="utf-8") return with path.open("w", newline="", encoding="utf-8") as f: writer = csv.DictWriter(f, fieldnames=list(rows[0].keys())) writer.writeheader() writer.writerows(rows)
def write_dot(path: Path, edges, graph_name, src_key, dst_key, label_key): lines = [f'digraph "{graph_name}" {{', "  rankdir=LR;", '  node [shape=box];'] for e in edges: src = str(e.get(src_key, "")).replace('"', '\"') dst = str(e.get(dst_key, "")).replace('"', '\"') label = str(e.get(label_key, "")).replace('"', '\"') lines.append(f'  "{src}" -> "{dst}" [label="{label}"];') lines.append("}") path.write_text("\n".join(lines), encoding="utf-8")
def build_import_edges(file_results): rows = [] for fr in file_results: for imp in fr["imports"]: rows.append( { "src_file": fr["file"], "edge_type": "import", "target": imp["module"], "lineno": imp["lineno"], } ) for imp in fr["from_imports"]: prefix = "." * int(imp.get("level", 0) or 0) mod = f"{prefix}{imp['module']}" if imp["module"] else prefix rows.append( { "src_file": fr["file"], "edge_type": "from_import", "target": f"{mod}:{imp['name']}", "lineno": imp["lineno"], } ) return rows
def build_call_edges(file_results): rows = [] for fr in file_results: rows.extend(fr["calls"]) return rows
def build_class_edges(file_results): rows = [] for fr in file_results: rows.extend(fr["class_edges"]) return rows
def build_centrality(import_edges, call_edges): score = Counter() for e in import_edges: score[e["src_file"]] += 1 for e in call_edges: score[e["src_file"]] += 1 return [ {"file": name, "score": value} for name, value in score.most_common() ]
def build_cycles(import_edges): deps = defaultdict(set) files = set() for e in import_edges: src = e["src_file"] dst = e["target"] files.add(src) deps[src].add(dst)
visited = set()
stack = []
onstack = set()
cycles = []

def dfs(node):
    visited.add(node)
    stack.append(node)
    onstack.add(node)

    for nei in deps.get(node, []):
        if nei not in visited:
            dfs(nei)
        elif nei in onstack:
            idx = stack.index(nei)
            cycles.append(stack[idx:] + [nei])

    stack.pop()
    onstack.remove(node)

for f in list(files):
    if f not in visited:
        dfs(f)

normalized = []
seen = set()
for cyc in cycles:
    key = tuple(cyc)
    if key not in seen:
        seen.add(key)
        normalized.append({"cycle": cyc})
return normalized
def main(): print("=== REPO AUTOPSY START ===") py_files = discover_python_files() print(f"Python files trovati: {len(py_files)}")
file_results = []

for p in py_files:

    else:
        print(f"❌ FAIL | {rel}")
        if result["syntax_error"]:
            se = result["syntax_error"]
            print(f"   SyntaxError line {se['lineno']} col {se['offset']}: {se['msg']}")
            if se.get("text"):
                print(f"   >>> {se['text'].rstrip()}")
        if result["traceback"]:
            print(result["traceback"])
        failed.append(rel)

import_edges = build_import_edges(file_results)
call_edges = build_call_edges(file_results)
class_edges = build_class_edges(file_results)
centrality = build_centrality(import_edges, call_edges)
cycles = build_cycles(import_edges)

write_json(OUT_DIR / "file_autopsy.json", file_results)
write_csv(OUT_DIR / "import_edges.csv", import_edges)
write_csv(OUT_DIR / "call_edges.csv", call_edges)

write_json(OUT_DIR / "centrality_ranking.json", centrality)

write_dot(OUT_DIR / "import_graph.dot", import_edges, "import_graph", "src_file", "target", "edge_type")

os.system(f'dot -Tpng "{OUT_DIR / "import_graph.dot"}" -o "{OUT_DIR / "import_graph.png"}"')

summary = {
    "python_files_total": len(py_files),

print("\n=== AUTOPSY SUMMARY ===")
print(json.dumps(summary, indent=2, ensure_ascii=False))

if failed:
    print("\n🚨 AUTOPSY FAIL")
    return 1

print("\n✅ AUTOPSY OK")
return 0
if name == "main": raise SystemExit(main())
